async def handle_document(update: Update, context: ContextTypes.DEFAULT_TYPE) -> None:
    try:
        file = await update.message.document.get_file()
        file_data = requests.get(file.file_path).content
        
        files = {"file": (update.message.document.file_name, file_data)}
        response = requests.post(FASTAPI_URL, files=files)
        
        if response.status_code == 200:
            await update.message.reply_text("âœ… Documento processado com sucesso!\n\n" + response.text)
        else:
            await update.message.reply_text(f"âŒ Erro no processamento: {response.text}")

    except Exception as e:
        await update.message.reply_text(f"âŒ Erro ao processar documento: {str(e)}")
